## S2.1. PsychoPy скрипт: 2AFC протокол с автоэкспортом

Скрипт реализует задачу последовательного предсказания с цепью Маркова 2-го порядка. Автоматически вычисляет `H_source` и `I_pred` по завершении блока и сохраняет данные в формате, готовом для `fit_delta_min.py`.

```python
# -*- coding: utf-8 -*-
"""
Protocol_2AFC_Prediction.py
Задача двухальтернативного выбора с последовательным предсказанием.
Экспорт данных совместим с fit_delta_min.py (cognitive-shadow)
Зависимости: psychopy, pandas, numpy
Версия: 2.0 (обновлена 8 июля 2026 г.)
"""

from psychopy import core, visual, event, gui, data, logging
import numpy as np
import pandas as pd
import os
import math
import time

# ================= КОНФИГУРАЦИЯ =================
BLOCK_ENTROPIES = [0.2, 0.35, 0.5, 0.65, 0.8, 1.0]  # Целевая условная энтропия (бит)
TRIALS_PER_BLOCK = 60
FIXATION_DUR = 0.5
PREDICT_DUR = 2.0
STIM_DUR = 0.5
FEEDBACK_DUR = 0.3
ITISI = 0.8

# Переходные вероятности для аппроксимации целевой энтропии (Марков-2)
TRANSITION_MAP = {
    0.2: {'LL': {'L': 0.85, 'R': 0.15}, 'RR': {'L': 0.15, 'R': 0.85}},
    0.35: {'LL': {'L': 0.75, 'R': 0.25}, 'RR': {'L': 0.25, 'R': 0.75}},
    0.5: {'LL': {'L': 0.65, 'R': 0.35}, 'RR': {'L': 0.35, 'R': 0.65}},
    0.65: {'LL': {'L': 0.55, 'R': 0.45}, 'RR': {'L': 0.45, 'R': 0.55}},
    0.8: {'LL': {'L': 0.51, 'R': 0.49}, 'RR': {'L': 0.49, 'R': 0.51}},
    1.0: {'LL': {'L': 0.5, 'R': 0.5}, 'RR': {'L': 0.5, 'R': 0.5}}
}

# ================= УТИЛИТЫ =================
def compute_conditional_entropy(sequence):
    """Вычисляет H(X_t | X_{t-1}, X_{t-2}) для бинарной последовательности."""
    if len(sequence) < 3: return 1.0
    counts = {'LL': {'L':0, 'R':0}, 'RR': {'L':0, 'R':0}, 'LR': {'L':0, 'R':0}, 'RL': {'L':0, 'R':0}}
    for i in range(2, len(sequence)):
        ctx = sequence[i-2] + sequence[i-1]
        nxt = sequence[i]
        if ctx in counts and nxt in counts[ctx]:
            counts[ctx][nxt] += 1
    ent = 0.0
    for ctx in counts:
        total = sum(counts[ctx].values())
        if total > 0:
            for nxt in counts[ctx]:
                p = counts[ctx][nxt] / total
                if p > 0: ent -= (total / (len(sequence)-2)) * p * math.log2(p)
    return ent

def export_fitting_data(df, subject_id, out_dir="data"):
    """Генерирует summary для fit_delta_min.py"""
    os.makedirs(out_dir, exist_ok=True)
    summary = []
    for target_h in BLOCK_ENTROPIES:
        block_data = df[df['block_target_entropy'] == target_h]
        actual_h = compute_conditional_entropy(block_data['stimulus'].tolist())
        acc = block_data['correct'].mean()
        h_emp = -acc*math.log2(acc) - (1-acc)*math.log2(1-acc) if 0<acc<1 else 0
        i_pred = actual_h - h_emp
        summary.append({
            'H_source': actual_h,
            'I_pred': max(0.0, min(1.0, i_pred)),
            'target_H': target_h,
            'n_trials': len(block_data)
        })
    out_df = pd.DataFrame(summary)
    out_path = os.path.join(out_dir, f"fitting_input_{subject_id}.csv")
    out_df.to_csv(out_path, index=False)
    return out_path

# ================= ЭКСПЕРИМЕНТ =================
def run_experiment():
    dlg = gui.DlgFromDict(dictionary={'participant': '', 'session': '001'}, title="cognitive-shadow 2AFC Protocol")
    if not dlg.OK: core.quit()
    subject = dlg.data['participant']
    win = visual.Window([1024, 768], color='gray', units='height', fullscr=False)
    fixation = visual.TextStim(win, text='+', height=0.1, color='white')
    stim_L = visual.Circle(win, radius=0.1, pos=(-0.2, 0), fillColor='blue')
    stim_R = visual.Circle(win, radius=0.1, pos=(0.2, 0), fillColor='red')
    feedback_txt = visual.TextStim(win, height=0.08, color='white')
    
    all_data = []
    np.random.seed(int(time.time()))
    
    for target_h in BLOCK_ENTROPIES:
        trans = TRANSITION_MAP[target_h]
        seq = ['L', 'L']
        for _ in range(TRIALS_PER_BLOCK + 2):
            ctx = seq[-2] + seq[-1]
            probs = trans.get(ctx, {'L': 0.5, 'R': 0.5})
            nxt = np.random.choice(['L', 'R'], p=[probs['L'], probs['R']])
            seq.append(nxt)
        
        for trial_idx, stim_code in enumerate(seq[2:]):
            fixation.draw()
            win.flip()
            core.wait(FIXATION_DUR)
            
            clock = core.Clock()
            keys = event.waitKeys(maxWait=PREDICT_DUR, keyList=['left', 'right'], timeStamped=clock)
            if not keys: resp, rt = 'none', 0
            else: resp, rt = keys[0]
            
            (stim_L if stim_code == 'L' else stim_R).draw()
            win.flip()
            core.wait(STIM_DUR)
            
            correct = (resp == f"{'left' if stim_code=='L' else 'right'}")
            feedback_txt.text = "Верно!" if correct else "Нет"
            feedback_txt.color = 'green' if correct else 'red'
            feedback_txt.draw()
            win.flip()
            core.wait(FEEDBACK_DUR)
            
            all_data.append({
                'subject': subject,
                'block': len([b for b in BLOCK_ENTROPIES if b < target_h]),
                'block_target_entropy': target_h,
                'trial': trial_idx,
                'response': resp,
                'stimulus': stim_code,
                'correct': int(correct),
                'RT': rt
            })
    
    df = pd.DataFrame(all_data)
    raw_path = os.path.join("data", f"raw_trials_{subject}.csv")
    df.to_csv(raw_path, index=False)
    fit_path = export_fitting_data(df, subject)
    
    print(f"✅ Данные сохранены: {raw_path}")
    print(f"📊 Данные для фитинга: {fit_path}")
    win.close()
    core.quit()

if __name__ == "__main__":
    run_experiment()
```

## S2.2. Протокол 1: Фальсификация шумодоминирования (Теорема 6)

**Проверяемый пробел:** Контрпродуктивный эффект интенции при сильном дисбалансе интерфейсов.

**Количественное обоснование (M ≈ 10⁴, δ_min ≈ 7×10⁻⁵).** Порог дисбаланса Δ_threshold ≈ 0.15–0.20 (нормированные единицы). При разности `C_refl − C_sens ≥ Δ_threshold` дополнительное усилие в сенсорном канале должно приводить к падению d′ на 0.3–0.5 (эффект Коэна d ≈ 0.5–0.8). Необходимый размер выборки: N = 25–35.

**Экспериментальная гипотеза:**  
При превышении порогового дисбаланса `Δ_threshold` увеличение волевого усилия (intent) в слабом сенсорном канале ведёт к *снижению* его производительности и избирательности (рост шума).

**Дизайн:** Внутригрупповой, многофакторный (2×2×2):  
- Фактор 1: Уровень рефлексивной нагрузки (низкий / высокий, последний должен создавать дисбаланс через перегрузку `C_refl`).  
- Фактор 2: Инструкция на усилие в зондовой сенсорной задаче («работайте в удобном темпе» / «выложитесь максимально»).  
- Фактор 3: Наличие шума (высокий уровень дистракторов в зондовой задаче).  

**Участники:** N = 40 здоровых взрослых (20–40 лет), предварительно откалиброванных по профилю интерфейсов для обеспечения вариативности `C_refl` и `R_max`.

**Задачи:**
- *Фоновая рефлексивная нагрузка:* адаптивная задача на оценку уверенности в перцептивных решениях (мета‑d′), сложность регулируется для поддержания ~80% точности. Высокая нагрузка дополнительно включает параллельное удержание в памяти 5-значного числа.
- *Зондовая сенсорная задача:* детекция слабого тонального сигнала в белом шуме; 80 проб за блок. В условиях «высокого шума» добавляется нерелевантный речевой поток.

**Процедура (1 сессия, 2 часа):**
1. Калибровка `M` (10 мин ЭЭГ покоя с открытыми глазами, расчёт Higuchi FD).
2. Базальное измерение мета‑d′ в обеих задачах по отдельности.
3. 8 экспериментальных блоков (условия рандомизированы), каждый по 5 мин. Внутри блока — совместное выполнение фоновой и зондовой задач.
4. В конце каждого блока — замер кортизола в слюне (проба Salivette) и субъективная оценка усилия (шкала Borg CR-10).

**Измеряемые переменные:**
- *Поведенческие:* d′ сенсорной задачи, время реакции, дисперсия ошибок.
- *Метакогнитивные:* мета‑d′ в фоновой и зондовой задачах.
- *Физиологические:* кортизол слюны (прокси `noise_k`), мощность альфа-ритма в зонах RLPFC (прокси `C_refl`).
- *Расчётные:* индивидуальный `Δ_threshold` из калибровочных данных (Приложение П6). Текущий дисбаланс оценивается как разность `C_refl` и `C_sens`, нормированная на `R_max`.

**Предсказание модели cognitive-shadow:**
В условиях, где `C_refl − C_sens ≥ Δ_threshold`:
- Прирост d′ от инструкции «стараться» будет отрицательным (d′_high_effort < d′_low_effort).
- Дисперсия ошибок возрастёт.
- Корреляция между субъективным усилием и d′ в этих условиях станет отрицательной.
- Эффект не будет наблюдаться при низкой рефлексивной нагрузке (дисбаланс отсутствует).

**Критерий фальсификации теоремы:**
Если ни в одном из условий не обнаружится значимого взаимодействия «Нагрузка × Инструкция», при котором высокое усилие ухудшает производительность (p > 0.05 при ANOVA с апостериорным тестом), то гипотеза шумодоминирования отвергается. В частности, если во всех условиях d′_high_effort ≥ d′_low_effort, модель cognitive-shadow в сильной форме фальсифицирована.

## S2.3. Протокол 2: Эмпирическая валидация предела управляемости (Теорема 7)

**Проверяемый пробел:** Связь между неустранимой остаточной стоимостью переключения (`τ_min`) и размерностной сложностью системы (`M`).

**Количественное обоснование:** При M ~10⁴ ожидаемый коэффициент корреляции Пирсона r ≈ 0.4–0.6 между δ_min и асимптотическим switch cost. Для обнаружения r=0.4 с мощностью 0.8 требуется N=47.

**Экспериментальная гипотеза:**  
Индивидуальная асимптота остаточной стоимости переключения задач (switch cost) положительно коррелирует с оценкой `δ_min(M) = ln 2 / M`, где `M` измеряется через фрактальную размерность Higuchi ЭЭГ.

**Дизайн:** Корреляционно-экспериментальный, две сессии.

**Участники:** N = 60 здоровых взрослых (широкий возрастной диапазон 20–65 лет для обеспечения вариативности `M`).

**Задача:** Две простые задачи на категоризацию: «чётность цифры» и «больше/меньше 5» для цифр 1–9 (кроме 5). Переключение происходит по внешнему сигналу (цвет рамки).

**Процедура:**
1. *Сессия 1 (ЭЭГ):* Запись 10 мин покоя с открытыми глазами (64 канала, ≥500 Гц). Расчёт Higuchi FD для каждого канала; усреднение по лобно-центральным отведениям даёт оценку `M` через калибровочную формулу `M = (R/ε)^D` с фиксированными параметрами окна.
2. *Сессия 2 (Поведенческая):* Тренировка до стабильной производительности (5 блоков по 120 проб). Затем измерение switch cost при разных интервалах подготовки (CTI: 100, 200, 400, 800, 1500 мс) — 10 блоков на каждый CTI, порядок случайный. Инструкция: «максимальная скорость при сохранении точности». Регистрируется время реакции (RT) для повторяющихся и переключаемых проб.

**Измеряемые переменные:**
- *Асимптотический switch cost:* вычисляется как разница RT между переключаемыми и повторяющимися пробами, усреднённая для наиболее длинных CTI (800 и 1500 мс), где достигается плато.
- *Индивидуальная оценка `δ_min`* через `M`, полученную на сессии 1.

**Предсказание cognitive-shadow:**
- Значимая положительная корреляция (r > 0.3, p < 0.05) между асимптотическим switch cost и `δ_min` (или отрицательная корреляция с `M`).
- Отсутствие корреляции между switch cost при коротких CTI и `δ_min` (поскольку при коротких CTI лимитирует не `τ_min`, а процессы реконфигурации).

**Критерий фальсификации:**  
Если корреляция между асимптотическим switch cost и `δ_min` статистически не значима (p > 0.1) и байесовский фактор указывает на нулевую модель (BF₀₁ > 3), то гипотеза о связи `τ_min` с `M` отвергается.

## S2.4. Протокол 3: Верификация порога необратимой деградации (Теорема 8)

**Проверяемый пробел:** Существование порога `Δ_crit` кумулятивной нагрузки, разделяющего обратимое утомление и необратимую деградацию `φ(refl)`.

**Количественное обоснование (M ≈ 10⁴, δ_min ≈ 7×10⁻⁵).** Δ_crit ≈ 0.05–0.10 бита, что соответствует 7–14 дням выраженного дисбаланса. Ожидаемый остаточный дефицит δ_perm ≥ 0.2–0.4 единицы meta‑d′ для участников выше порога.

**Экспериментальная гипотеза:**  
У участников, накопивших кумулятивную нагрузку `Load ≥ Δ_crit`, после длительного отдыха сохранится остаточный дефицит метакогнитивной точности `δ_perm > 0`, в то время как при `Load < Δ_crit` дефицит исчезнет полностью.

**Дизайн:** Лонгитюдный, с двумя группами, формируемыми post-hoc по медиане накопленной нагрузки.

**Участники:** N = 50 здоровых студентов в период экзаменационной сессии (естественная высокая нагрузка) или участники, проходящие 3-недельный интенсивный учебный курс с контролируемой нагрузкой.

**Процедура (6 недель):**
- *Неделя 0:* Базовые измерения: мета‑d′ (перцептивное различение + уверенность), опросник выгорания (MBI), кортизол (утренний), ВСР, полисомнография (одна ночь).
- *Недели 1–3 (Нагрузочная фаза):* Ежедневные 8-часовые занятия, требующие высокой концентрации и решения сложных задач; вечерние когнитивные тесты; ограничение сна (5–6 ч). Еженедельный замер мета‑d′, MBI, кортизола.
- *Недели 4–5 (Фаза отдыха):* Полное прекращение учебной нагрузки, свободный режим, поощрение сна.
- *Неделя 6:* Итоговые измерения, идентичные неделе 0.

**Расчёт кумулятивной нагрузки:**
Для каждого участника на каждой неделе вычисляется мгновенное отклонение `ΔE(t)` от базового нормального профиля. Профиль задаётся тремя показателями: мета‑d′, утренний кортизол, RMSSD. Нормализованные значения z-трансформируются, и `ΔE(t)` есть евклидово расстояние от базового вектора. Кумулятивная нагрузка `Load` = сумма `ΔE(t)` за недели 1–3.

**Определение порога `Δ_crit`:** Методом кусочно-линейной регрессии (segmented regression) определяется точка перегиба на кривой зависимости остаточного дефицита `δ_perm` (разница мета‑d′ Неделя 0 − Неделя 6) от `Load`.

**Предсказание модели cognitive-shadow:**
- Существует статистически значимый порог `Δ_crit`, ниже которого `δ_perm` не отличается от нуля, а выше — `δ_perm > 0` и линейно растёт с `Load`.
- Группа выше порога будет демонстрировать метакогнитивный парадокс: субъективные жалобы на усталость (MBI) восстановятся до базового уровня, а объективный `δ_perm` сохранится.

**Критерий фальсификации:**
Если кусочно-линейная модель с порогом не даёт значимо лучшего объяснения данных по сравнению с линейной моделью без порога (тест Дэвиса, p > 0.1), или если после отдыха `δ_perm` не отличается от нуля у всех участников независимо от `Load`, то Теорема 8 фальсифицирована в части необратимости.

## S2.5. Протокол 4: Прямая проверка метакогнитивного коллапса (Теорема 9)

**Проверяемый пробел:** Существование порога `φ_crit`, ниже которого интенция становится вредной и самовосстановление невозможно без внешней помощи.

**Количественное обоснование:** Для здоровых φ(refl) соответствует meta‑d′ 1.5–2.5. Порог коллапса φ_crit ≈ 0.5–0.8 meta‑d′. Ожидаемый отрицательный эффект усилия: падение на 0.3–0.6.

**Экспериментальная гипотеза:**
1. У лиц с `φ(refl) < φ_crit` инструкция «стараться сильнее» в метакогнитивной задаче *снизит* метакогнитивную точность.
2. После стрессорного воздействия у них будет наблюдаться лавинообразное падение `φ(refl)`, которое не прекратится после отдыха, но остановится при оказании внешней структурированной поддержки.

**Дизайн:** Смешанный, с тремя группами сравнения и контролем.

**Участники:**
- Группа 1 (n=20): Здоровый контроль с высоким `φ(refl)` (мета‑d′ > 1.5).
- Группа 2 (n=20): Субклиническая депрессия (PHQ‑9 10–14), средний `φ(refl)`.
- Группа 3 (n=20): Клиническая депрессия (PHQ‑9 ≥ 15), низкий `φ(refl)` (< 1.0).

**Процедура (1 сессия, 3 часа):**
1. **Базовый замер:** Мета‑d′ в задаче различения (100 проб), шкала настроения (PANAS).
2. **Стресс-индукция:** Трирский социальный стресс-тест (TSST) — 5 мин подготовки, 5 мин речи, 5 мин арифметики перед комиссией. Сразу после — замер кортизола и мета‑d′.
3. **Фаза «Усилия»:** Повторное выполнение метакогнитивной задачи с инструкцией «Вы должны превзойти свой предыдущий результат; от этого зависит важный бонус».
4. **Фаза спонтанного отдыха:** 40 мин в тихой комнате без какой-либо стимуляции. Замер мета‑d′.
5. **Фаза внешней поддержки:** Экспериментатор предоставляет стандартизированную обратную связь («Ваш результат теперь стабилен, вы хорошо справляетесь»), даёт простую мнемоническую стратегию для оценки уверенности, затем ещё один замер мета‑d′.

**Измерения:** Мета‑d′ на всех этапах, кортизол после TSST и в конце отдыха, частота сердечных сокращений.

**Предсказание модели cognitive-shadow:**
- В Группе 3:
  - `мета‑d′_усилия < мета‑d′_после TSST` (отрицательный эффект усилия).
  - `мета‑d′_отдых ≤ мета‑d′_после TSST` (отсутствие спонтанного восстановления).
  - `мета‑d′_поддержка > мета‑d′_отдых` (восстановление только при внешнем оркестраторе).
- В Группе 1 и 2: усилие либо повышает, либо не меняет мета‑d′; после отдыха происходит спонтанное восстановление к базовому уровню.
- Точка `φ_crit` оценивается post-hoc как пороговое значение `φ(refl)` на этапе 2, ниже которого начинает наблюдаться отрицательный эффект усилия.

**Критерий фальсификации:**
Если в Группе 3 (низкий `φ(refl)`) не обнаружится значимого снижения мета‑d′ в фазе усилия или если после отдыха все группы продемонстрируют восстановление без внешней помощи, то предсказания Теоремы 9 о пороге коллапса и необходимости внешнего оркестратора считаются фальсифицированными.

## S2.7. Связь с формальной верификацией FORCED_REPORT и Shadow Ethics

Все протоколы разработаны с учётом следующих формально верифицированных инвариантов:

1. **Запрет на провокацию коллапса (Принцип 2 Shadow Ethics):** Ни один протокол не предполагает намеренное введение испытуемого в состояние `φ(refl) ≤ φ_crit`. Работа ведётся только с естественной вариативностью или уже существующими клиническими состояниями.

2. **Обязательство внешней поддержки (Принцип 3):** В протоколе 4 фаза внешней поддержки является обязательной частью процедуры, а не факультативным дополнением.

3. **Звуковость FORCED_REPORT:** Протокол FORCED_REPORT формально верифицирован в Coq (модуль `ForcedReportDecision.v`) — переход в состояние `CONSCIOUS_LOCKED` происходит только при выполнении всех триггеров, что гарантирует отсутствие ложноположительных диагнозов.

4. **Этическая консистентность:** Все протоколы согласованы с шестью принципами Shadow Ethics, формально верифицированными в `FormalEthicsPrinciples.v`.
